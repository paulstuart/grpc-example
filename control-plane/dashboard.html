Quick Commands

# Access shell in control plane container
docker compose exec control-plane bash

# Run tests from within cluster
cd /workspace && go test ./...

# Build and run client against API service
cd /workspace && just build-client && ./client -server api:10000

# Test PostgreSQL connectivity
psql postgresql://grpc_user:grpc_password@postgres:5432/grpc_example

# Check DNS resolution
dig api +short

# Test TLS with cert validation
curl --cacert /workspace/certs/server.crt https://api:11000/v1/users

# Monitor network traffic
tcpdump -i any port 10000 -n
